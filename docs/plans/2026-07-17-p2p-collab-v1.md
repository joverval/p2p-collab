> **For Hermes:** Use subagent-driven-development skill to implement this plan task-by-task.

**Goal:** Build a browser-only P2P collaboration library (`@joverval/p2p-collab`) that enables WebRTC data channel connections using only URL-encoded SDP exchange, plus a markdown collaborative editor as the first app on top.

**Architecture:** Two layers. **Base library** (Phase 1): TypeScript package using simple-peer for WebRTC, host-star topology, URL fragment signaling. Exports `createRoom()` / `joinRoom(url)`. **Markdown app** (Phase 2): vanilla HTML/TS app using Yjs CRDT over the library's data channel, with File System Access API for local file persistence on the host.

**Tech Stack:** TypeScript, simple-peer@9.11.1, tsup, vitest, Yjs, File System Access API

```typescript
// src/signal.ts
import type { SignalData } from './types';

/**
 * Encode WebRTC signal data into a sharable URL.
 * Uses URL fragment (#sdp=...) so the encoded SDP is never sent to a server.
 */
export function encodeSignal(
  data: SignalData,
  baseUrl: string = ''
): { url: string; sizeKB: number } {
  const json = JSON.stringify(data);
  const encoded = btoa(json);

  // Build URL: if baseUrl is provided, append fragment to it.
  // Otherwise return just the fragment. Strip existing fragments.
  const base = baseUrl.split('#')[0];
  const url = base ? `${base}#sdp=${encoded}` : `#sdp=${encoded}`;

  return {
    url,
    sizeKB: parseFloat((encoded.length / 1024).toFixed(1)),
  };
}

/**
 * Decode a signal URL or raw base64 string back to SignalData.
 * Returns null if the input is invalid or can't be decoded.
 */
export function decodeSignal(input: string): SignalData | null {
  if (!input) return null;

  try {
    // Extract the base64 payload: either from #sdp= fragment or treat whole string as b64
    const fragment = input.includes('#sdp=')
      ? input.split('#sdp=')[1]
      : input;

    const json = atob(fragment);
    const parsed: SignalData = JSON.parse(json);

    // Must be an object with at least one known WebRTC signal field
    if (!parsed || typeof parsed !== 'object') return null;
    if (!parsed.type && !parsed.sdp && !parsed.candidate) return null;

    return parsed;
  } catch {
    return null;
  }
}
```

```typescript
// src/room.ts
import SimplePeer from 'simple-peer';
import { encodeSignal, decodeSignal } from './signal';
import type { Room, RoomOptions, PeerInfo, SignalData } from './types';

type MessageHandler = (data: string | Uint8Array, peerId: string) => void;
type PeerJoinHandler = (peerId: string) => void;

let peerCounter = 0;

export class P2PRoom implements Room {
  public readonly isHost: boolean;

  // Host state
  private _hostPeer?: SimplePeer.Instance;  // the "pending" peer awaiting answer
  private _peers: Map<string, SimplePeer.Instance> = new Map();
  private _peerInfos: PeerInfo[] = [];

  // Peer state
  private _peer?: SimplePeer.Instance;

  // Handlers
  private _onMessage: MessageHandler | null = null;
  private _onPeerJoin: PeerJoinHandler | null = null;
  private _onPeerLeave?: (peerId: string) => void;
  private _onError?: (err: Error) => void;
  private _onClose?: () => void;

  // Base URL for signal encoding
  private _baseUrl: string;

  // Promise that resolves when the host offer is ready
  private _offerPromise: Promise<string>;

  constructor(isHost: boolean, baseUrl: string, opts: RoomOptions = {}) {
    this.isHost = isHost;
    this._baseUrl = baseUrl;
    this._onPeerLeave = opts.onPeerLeave;
    this._onError = opts.onError;
    this._onClose = opts.onClose;

    if (isHost) {
      this._offerPromise = this._createHostOffer();
    } else {
      this._offerPromise = Promise.reject(new Error('joinRoom must be used for peers'));
    }
  }

  /** Wait for the offer URL to be generated. For host only. */
  offerUrl(): Promise<string> {
    return this._offerPromise;
  }

  /** Accept a peer's answer signal. Host only. */
  acceptAnswer(signalUrl: string): void {
    const data = decodeSignal(signalUrl);
    if (!data || !this._hostPeer) {
      this._onError?.(new Error('Invalid answer signal or no pending connection'));
      return;
    }
    this._hostPeer.signal(data);
  }

  /** Connect to host using offer URL. Peer only. */
  connectToHost(offerUrl: string): void {
    if (this.isHost) {
      this._onError?.(new Error('Host cannot connectToHost'));
      return;
    }

    const signalData = decodeSignal(offerUrl);
    if (!signalData) {
      this._onError?.(new Error('Invalid offer URL'));
      return;
    }

    this._createPeer(false, (url) => {
      // Peer generates an answer — we expose it but the user must deliver it
      // to the host out-of-band. We store for retrieval.
      this._pendingAnswer = url;
    });

    this._peer!.signal(signalData);
  }

  private _pendingAnswer?: string;

  /** Get the answer URL after connectToHost. Peer only. */
  getAnswerUrl(): string | undefined {
    return this._pendingAnswer;
  }

  /** Force a peer to generate an answer (used after connectToHost). */
  generateAnswer(): string | undefined {
    return this._pendingAnswer;
  }

  // ---- Public API ----

  get peers(): PeerInfo[] {
    return this._peerInfos;
  }

  send(data: string | Uint8Array): void {
    if (this.isHost) {
      // Broadcast to all peers
      for (const peer of this._peers.values()) {
        peer.send(data);
      }
    } else if (this._peer?.connected) {
      this._peer.send(data);
    }
  }

  onMessage(handler: MessageHandler): void {
    this._onMessage = handler;
  }

  onPeerJoin(handler: PeerJoinHandler): void {
    this._onPeerJoin = handler;
  }

  close(): void {
    if (this._hostPeer) {
      this._hostPeer.destroy();
      this._hostPeer = undefined;
    }
    if (this._peer) {
      this._peer.destroy();
      this._peer = undefined;
    }
    for (const p of this._peers.values()) {
      p.destroy();
    }
    this._peers.clear();
    this._peerInfos = [];
    this._onClose?.();
  }

  // ---- Internal ----

  private async _createHostOffer(): Promise<string> {
    return new Promise((resolve, reject) => {
      const peer = new SimplePeer({ initiator: true, trickle: false });

      peer.on('signal', (data: SignalData) => {
        const { url } = encodeSignal(data, this._baseUrl);
        // Store as the next pending peer connection
        this._hostPeer = peer;
        resolve(url);
      });

      peer.on('connect', () => {
        const peerId = `peer-${++peerCounter}`;
        this._peers.set(peerId, peer);
        this._peerInfos.push({
          id: peerId,
          send: (d: string | Uint8Array) => peer.send(d),
        });
        this._onPeerJoin?.(peerId);

        // Set up data handler
        peer.on('data', (data: Uint8Array) => {
          const text = new TextDecoder().decode(data);
          this._onMessage?.(text, peerId);
        });

        peer.on('close', () => {
          this._peers.delete(peerId);
          this._peerInfos = this._peerInfos.filter(p => p.id !== peerId);
          this._onPeerLeave?.(peerId);
        });

        // Ready for next peer
        this._hostPeer = undefined;
      });

      peer.on('error', (err: Error) => {
        this._onError?.(err);
        reject(err);
      });
    });
  }

  private _createPeer(
    initiator: boolean,
    onAnswer: (url: string) => void
  ): void {
    const peer = new SimplePeer({ initiator, trickle: false });

    peer.on('signal', (data: SignalData) => {
      const { url } = encodeSignal(data, this._baseUrl);
      onAnswer(url);
    });

    peer.on('connect', () => {
      this._onMessage
        ? peer.on('data', (data: Uint8Array) => {
            const text = new TextDecoder().decode(data);
            this._onMessage?.(text, 'host');
          })
        : null;
    });

    peer.on('error', (err: Error) => {
      this._onError?.(err);
    });

    peer.on('close', () => {
      this._onClose?.();
    });

    this._peer = peer;
  }
}
```

```typescript
// src/signal.ts
import type { SignalData } from './types';

export function encodeSignal(
  data: SignalData,
  baseUrl: string = ''
): { url: string; sizeKB: number } {
  const json = JSON.stringify(data);
  const encoded = btoa(json);
  const base = baseUrl.split('#')[0];
  const url = base
    ? `${base}#sdp=${encoded}`
    : `#sdp=${encoded}`;
  return { url, sizeKB: +(encoded.length / 1024).toFixed(1) };
}

export function decodeSignal(input: string): SignalData | null {
  if (!input) return null;
  try {
    const fragment = input.includes('#sdp=')
      ? input.split('#sdp=')[1]
      : input;
    const parsed = JSON.parse(atob(fragment));
    if (!parsed || typeof parsed !== 'object') return null;
    if (!parsed.type && !parsed.sdp && !parsed.candidate) return null;
    return parsed as SignalData;
  } catch {
    return null;
  }
}
```

```typescript
// src/room.ts
import SimplePeer from 'simple-peer';
import { encodeSignal, decodeSignal } from './signal';
import type { Room, RoomOptions, PeerInfo, SignalData, CreateRoomResult } from './types';

let peerCounter = 0;

export class P2PRoom implements Room {
  public readonly isHost: boolean;

  // Host state
  private _hostOfferPeer?: SimplePeer.Instance;
  private _peers: Map<string, SimplePeer.Instance> = new Map();
  private _peerInfos: PeerInfo[] = [];
  private _offerResolve?: (url: string) => void;

  // Peer state
  private _peer?: SimplePeer.Instance;
  private _answerResolve?: (url: string) => void;

  // Handlers
  private _onMessage?: (data: string | Uint8Array, peerId: string) => void;
  private _onPeerJoin?: (peerId: string) => void;
  private readonly _onPeerLeave?: (peerId: string) => void;
  private readonly _onError?: (err: Error) => void;
  private readonly _onClose?: () => void;

  private readonly _baseUrl: string;

  constructor(isHost: boolean, baseUrl: string, opts: RoomOptions = {}) {
    this.isHost = isHost;
    this._baseUrl = baseUrl;
    this._onPeerLeave = opts.onPeerLeave;
    this._onError = opts.onError;
    this._onClose = opts.onClose;
  }

  // ── Host API ──

  /** Returns a promise that resolves with the offer URL. Host only. */
  offerUrl(): Promise<string> {
    if (!this.isHost) return Promise.reject(new Error('Only host can generate offers'));
    return new Promise((resolve, reject) => {
      const peer = new SimplePeer({ initiator: true, trickle: false });
      this._hostOfferPeer = peer;

      peer.on('signal', (data: SignalData) => {
        const { url } = encodeSignal(data, this._baseUrl);
        resolve(url);
      });

      peer.on('connect', () => this._onPeerConnected(peer));
      peer.on('error', (err: Error) => {
        this._onError?.(err);
        reject(err);
      });
    });
  }

  /** Accept a peer's answer signal. Host only. */
  acceptAnswer(signalUrl: string): void {
    if (!this.isHost || !this._hostOfferPeer) {
      this._onError?.(new Error('No pending connection'));
      return;
    }
    const data = decodeSignal(signalUrl);
    if (!data) {
      this._onError?.(new Error('Invalid answer URL'));
      return;
    }
    this._hostOfferPeer.signal(data);
  }

  // ── Peer API ──

  /**
   * Connect to a host using their offer URL. Peer only.
   * Returns a promise that resolves with the answer URL.
   */
  connectToHost(offerUrl: string): Promise<string> {
    if (this.isHost) return Promise.reject(new Error('Host cannot connectToHost'));
    const signalData = decodeSignal(offerUrl);
    if (!signalData) return Promise.reject(new Error('Invalid offer URL'));

    return new Promise((resolve, reject) => {
      const peer = new SimplePeer({ initiator: false, trickle: false });

      peer.on('signal', (data: SignalData) => {
        const { url } = encodeSignal(data, this._baseUrl);
        resolve(url);
      });

      peer.on('connect', () => {
        peer.on('data', (data: Uint8Array) => {
          const text = new TextDecoder().decode(data);
          this._onMessage?.(text, 'host');
        });
      });

      peer.on('error', (err: Error) => {
        this._onError?.(err);
        reject(err);
      });

      peer.on('close', () => this._onClose?.());

      this._peer = peer;
      peer.signal(signalData);
    });
  }

  // ── Shared API ──

  get peers(): PeerInfo[] {
    return this._peerInfos;
  }

  send(data: string | Uint8Array): void {
    if (this.isHost) {
      for (const peer of this._peers.values()) {
        peer.send(data);
      }
    } else if (this._peer) {
      this._peer.send(data);
    }
  }

  onMessage(handler: (data: string | Uint8Array, peerId: string) => void): void {
    this._onMessage = handler;
  }

  onPeerJoin(handler: (peerId: string) => void): void {
    this._onPeerJoin = handler;
  }

  close(): void {
    this._hostOfferPeer?.destroy();
    for (const p of this._peers.values()) p.destroy();
    this._peer?.destroy();
    this._peers.clear();
    this._peerInfos = [];
    this._onClose?.();
  }

  // ── Internal ──

  private _onPeerConnected(peer: SimplePeer.Instance): void {
    const peerId = `peer-${++peerCounter}`;
    this._peers.set(peerId, peer);
    this._peerInfos.push({
      id: peerId,
      send: (d: string | Uint8Array) => peer.send(d),
    });
    this._onPeerJoin?.(peerId);

    peer.on('data', (data: Uint8Array) => {
      const text = new TextDecoder().decode(data);
      this._onMessage?.(text, peerId);
    });

    peer.on('close', () => {
      this._peers.delete(peerId);
      this._peerInfos = this._peerInfos.filter(p => p.id !== peerId);
      this._onPeerLeave?.(peerId);
    });

    // Reset host offer peer so next acceptAnswer creates a new one
    this._hostOfferPeer = undefined;
  }
}
```

**Objective:** Replace manual `send()` with Yjs document sync. Host creates Y.Doc; peers receive it via the library's data channel.

**Approach:** Instead of raw message passing, serialize Yjs updates and transport them via `room.send()` / `room.onMessage()`.

```typescript
// On each peer connection, sync initial state
room.onPeerJoin((peerId) => {
```

Wait — this naive approach sends the full Yjs update on every keystroke. The proper approach: listen to Yjs `update` events and send those deltas.

```typescript
// Host only: listen for Yjs updates and broadcast

  room.send(update);
});

// Both: apply incoming updates
room.onMessage((data, peerId) => {
  if (data instanceof Uint8Array) {
    Y.applyUpdate(ydoc, data);
  }
});
```

But this requires that the host also sends the full initial state to newly connected peers. Use `onPeerJoin` to send `Y.encodeStateAsUpdate(ydoc)`.

```typescript
async function openFile() {
  const [handle] = await window.showOpenFilePicker({
    types: [{ accept: { 'text/markdown': ['.md'] } }],
  });
  fileHandle = handle;
  const file = await handle.getFile();
  const content = await file.text();
  ydoc.transact(() => {
    ytext.delete(0, ytext.length);
    ytext.insert(0, content);
  });
}

async function saveFile() {
  if (!fileHandle) return;
  const writable = await fileHandle.createWritable();
  await writable.write(ytext.toString());
  await writable.close();
}
```
